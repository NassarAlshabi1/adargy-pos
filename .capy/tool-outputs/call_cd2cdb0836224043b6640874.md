   178	  void _restoreData() async {
   179	    try {
   180	      // استخدام FilePicker لاختيار ملف النسخة الاحتياطية
   181	      FilePickerResult? result = await FilePicker.platform.pickFiles(
   182	        type: FileType.custom,

   184	        dialogTitle: 'اختر ملف النسخة الاحتياطية',

   187	      if (result == null || result.files.isEmpty) {
   188	        // المستخدم ألغى اختيار الملف
   189	        return;
   190	      }
   191	
   192	      // قراءة محتوى الملف
   193	      final filePath = result.files.single.path;
   194	      if (filePath == null) {
   195	        throw Exception('مسار الملف غير صالح');
   196	      }

   198	      final file = File(filePath);

   222	              child: Text('استعادة', style: TextStyle(fontFamily: 'Roboto')),
   223	            ),
   224	          ],
   225	          actionsAlignment: MainAxisAlignment.spaceBetween,
   226	        ),
   227	      );
   228	
   229	      if (confirmRestore != true) {
   230	        return;
   231	      }
   232	
   233	      // تحليل البيانات
   234	      Map<String, dynamic> backupData;
   235	      try {
   236	        backupData = json.decode(jsonString);
   237	      } catch (e) {
   238	        // محاولة التعامل مع النسخ الاحتياطية القديمة
   239	        final cleanedJsonString = jsonString.replaceAll(RegExp(r'\[|\]'), '');
   240	        final entriesJsonList = cleanedJsonString.split('},').map((entryStr) {
   241	          if (!entryStr.endsWith('}')) {
   242	            return '$entryStr}';

   244	          return entryStr;
   245	        }).toList();

   247	        backupData = {
   248	          'entries': entriesJsonList
   249	              .map((jsonStr) {
   250	                if (jsonStr.isEmpty) return null;
   251	                try {
   252	                  return json.decode(jsonStr);
   253	                } catch (e) {
   254	                  return null;
   255	                }
   256	              })
   257	              .where((item) => item != null)
   258	              .toList(),
   259	          'customers': [],
   260	          'transactions': [],
   261	          'settlements': [],
   262	        };
   263	      }
   264	
   265	      final entryProvider = Provider.of<EntryProvider>(context, listen: false);
   266	      final customerDbHelper = CustomerDatabaseHelper();
   267	
   268	      int restoredCount = 0;
   269	
   270	      // استعادة الإدخالات
   271	      if (backupData['entries'] != null) {
   272	        List<Entry> restoredEntries = [];
   273	        for (var entryData in backupData['entries']) {
   274	          try {
   275	            final entry = Entry.fromMap(entryData);
   276	            restoredEntries.add(entry);
   277	          } catch (e) {

   285	      // استعادة بيانات العملاء والمعاملات والسدادات
   286	      if (backupData['customers'] != null ||
   287	          backupData['transactions'] != null ||
   288	          backupData['settlements'] != null) {
   289	        final customers =
   290	            List<Map<String, dynamic>>.from(backupData['customers'] ?? []);
   291	        final transactions =
   292	            List<Map<String, dynamic>>.from(backupData['transactions'] ?? []);

   294	            List<Map<String, dynamic>>.from(backupData['settlements'] ?? []);

   656	  Widget _buildNavItem(int index, bool isDark) {
   657	    final bool isSelected = _currentIndex == index;
   658	    final _NavItemData item = _navItems[index];
   659	
   660	    final Color activeColor =
   661	        isDark ? AppTheme.secondaryColor : AppTheme.primaryColor;
   662	    final Color inactiveColor =
   663	        isDark ? Colors.grey.shade400 : AppTheme.lightSecondaryTextColor;
   664	
   665	    return Expanded(
   666	      child: Material(
   667	        color: Colors.transparent,
   668	        child: InkWell(
   669	          onTap: () {
   670	            setState(() {
   671	              _currentIndex = index;
   672	              _tabController.animateTo(index);
   673	            });
   674	          },
   675	          borderRadius: BorderRadius.circular(12),
   676	          splashColor: activeColor.withOpacity(0.1),
   677	          highlightColor: activeColor.withOpacity(0.05),
   678	          child: AnimatedContainer(
   679	            duration: const Duration(milliseconds: 200),
   680	            padding: const EdgeInsets.symmetric(vertical: 8, horizontal: 4),
   681	            decoration: BoxDecoration(
   682	              color: isSelected
   683	                  ? (isDark
   684	                      ? activeColor.withOpacity(0.15)
   685	                      : activeColor.withOpacity(0.08))
   686	                  : Colors.transparent,
   687	              borderRadius: BorderRadius.circular(12),
   688	            ),
   689	            child: Column(
   690	              mainAxisSize: MainAxisSize.min,
   691	              children: [
   692	                Icon(
   693	                  isSelected ? item.activeIcon : item.inactiveIcon,
   694	                  color: isSelected ? activeColor : inactiveColor,
   695	                  size: 24,
   696	                ),
   697	                const SizedBox(height: 4),
   698	                Text(
   699	                  item.label,
   700	                  style: TextStyle(
   701	                    color: isSelected ? activeColor : inactiveColor,
